
function Clamp(oldVal, min, max)
    if oldVal < min then
        return min
    elseif oldVal > max then
        return max
    else
        return oldVal
    end
end

function Vector2Distance(p1_x, p1_y, p2_x, p2_y)
    return math.sqrt((p2_x - p1_x) * (p2_x - p1_x) + (p2_y - p1_y) * (p2_y - p1_y));
end
